  <script>
    const fileInput = document.getElementById("fileInput");
    const uploadBtn = document.getElementById("uploadBtn");
    const errorEl = document.getElementById("error");
    const resultsEl = document.getElementById("results");
    const emptyStateEl = document.getElementById("emptyState");
    const depWeatherCard = document.getElementById("depWeatherCard");
    const destWeatherCard = document.getElementById("destWeatherCard");
    const altWeatherCard = document.getElementById("altWeatherCard");
    const depNotamsCard = document.getElementById("depNotamsCard");
    const destNotamsCard = document.getElementById("destNotamsCard");
    const altNotamsCard = document.getElementById("altNotamsCard");
    const detailsGrid = document.getElementById("detailsGrid");
    const dispatchNotesGrid = document.getElementById("dispatchNotesGrid");
    const statusPill = document.getElementById("statusPill");
    const statusCopy = document.getElementById("statusCopy");

    const summaryFlight = document.getElementById("summaryFlight");
    const summaryDate = document.getElementById("summaryDate");
    const summaryRoute = document.getElementById("summaryRoute");
    const summaryAirports = document.getElementById("summaryAirports");
    const summarySchedule = document.getElementById("summarySchedule");
    const summaryScheduleLocal = document.getElementById("summaryScheduleLocal");
    const summaryOfp = document.getElementById("summaryOfp");
    const summaryFlightTime = document.getElementById("summaryFlightTime");
    const summaryAircraft = document.getElementById("summaryAircraft");
    const summaryOps = document.getElementById("summaryOps");

    function safeValue(value, fallback = "—") {
      if (value === undefined || value === null || value === "") {
        return fallback;
      }
      return value;
    }

    function formatFlightTime(value) {
      const raw = safeValue(value, "").toString().trim();

      if (!raw) {
        return "—";
      }

      const match = raw.match(/^(\d{1,2}):(\d{2})$/);
      if (!match) {
        return raw;
      }

      const hours = parseInt(match[1], 10);
      const minutes = parseInt(match[2], 10);

      return `${hours}h ${minutes}m`;
    }

    function joinParts(parts, separator = " · ") {
      return parts.filter(Boolean).join(separator) || "—";
    }

    function resetUI() {
      errorEl.textContent = "";
      resultsEl.style.display = "none";
      emptyStateEl.style.display = "grid";
      detailsGrid.innerHTML = "";
      dispatchNotesGrid.innerHTML = "";
      depWeatherCard.innerHTML = "";
      destWeatherCard.innerHTML = "";
      altWeatherCard.innerHTML = "";
      depNotamsCard.innerHTML = "";
      destNotamsCard.innerHTML = "";
      altNotamsCard.innerHTML = "";
      statusPill.textContent = "Information only";
      statusCopy.textContent = "Info only. Always verify against the full OFP.";

      summaryFlight.textContent = "—";
      summaryDate.textContent = "—";
      summaryRoute.textContent = "—";
      summaryAirports.textContent = "—";
      summarySchedule.textContent = "—";
      summaryScheduleLocal.textContent = "—";
      summaryOfp.textContent = "—";
      summaryFlightTime.textContent = "—";
      summaryAircraft.textContent = "—";
      summaryOps.textContent = "—";
    }

    function createDataItem(label, value) {
      const item = document.createElement("div");
      item.className = "data-item";

      const labelEl = document.createElement("div");
      labelEl.className = "data-label";
      labelEl.textContent = label;

      const valueEl = document.createElement("div");
      valueEl.className = "data-value";
      valueEl.textContent = safeValue(value);

      item.appendChild(labelEl);
      item.appendChild(valueEl);

      return item;
    }

    function renderSummary(strip) {
      summaryFlight.textContent = joinParts([
        safeValue(strip.commercial_flight, "").trim(),
        safeValue(strip.atc_flight, "").trim()
      ], " / ");

      summaryDate.textContent = safeValue(strip.date_of_flight);

      summaryRoute.textContent = joinParts([
        strip.dep_iata || strip.dep_icao,
        strip.dest_iata || strip.dest_icao
      ], " → ");

      summaryAirports.textContent = joinParts([
        safeValue(strip.dep_rwy, "").trim() ? `DEP RWY ${strip.dep_rwy}` : "",
        safeValue(strip.arr_rwy, "").trim() ? `ARR RWY ${strip.arr_rwy}` : ""
      ]);

      summarySchedule.textContent = joinParts([
        safeValue(strip.std_utc, "").trim() ? `STD ${strip.std_utc}` : "",
        safeValue(strip.sta_utc, "").trim() ? `STA ${strip.sta_utc}` : ""
      ]);

      summaryScheduleLocal.textContent = joinParts([
        safeValue(strip.std_local, "").trim() ? `Local STD ${strip.std_local}` : "",
        safeValue(strip.sta_local, "").trim() ? `Local STA ${strip.sta_local}` : ""
      ]);

      summaryOfp.textContent = safeValue(strip.ofp_version);
      summaryFlightTime.textContent = formatFlightTime(strip.flight_time);

      summaryAircraft.textContent = safeValue(strip.reg);
      summaryOps.textContent = joinParts([
        safeValue(strip.engine_type, "").trim() ? `Engine ${strip.engine_type}` : ""
      ]);
    }

    function renderDetails(strip) {
      detailsGrid.innerHTML = "";

      const fields = [
        ["Departure", joinParts([strip.dep_name, strip.dep_icao, strip.dep_iata], " / ")],
        ["Destination", joinParts([strip.dest_name, strip.dest_icao, strip.dest_iata], " / ")],
        ["Alternate", joinParts([strip.alt_name, strip.alt_icao, strip.alt_iata], " / ")],
        ["FL range", joinParts([
          safeValue(strip.fl_min, "").trim() ? `Min ${strip.fl_min}` : "",
          safeValue(strip.fl_max, "").trim() ? `Max ${strip.fl_max}` : ""
        ])],
        ["Cost index", strip.cost_index],
        ["Ramp fuel", strip.ramp_fuel],
        ["EZFW", strip.ezfw],
        ["ETOW", strip.etow],
        ["ELWT", strip.elwt],
        ["Max shear", strip.max_shear_rate],
        ["MORA", strip.highest_mora]
      ];

      fields.forEach(([label, value]) => {
        detailsGrid.appendChild(createDataItem(label, value));
      });
    }

    function renderDispatchNotes(notes) {
      dispatchNotesGrid.innerHTML = "";

      const list = Array.isArray(notes) ? notes.filter(note => note && note.trim()) : [];

      if (!list.length) {
        dispatchNotesGrid.appendChild(createDataItem("Dispatch notes", "(none found)"));
        return;
      }

      list.forEach((note) => {
        dispatchNotesGrid.appendChild(createDataItem("Dispatch note", note));
      });
    }

    function renderWeatherCard(el, titleText, wx) {
      el.innerHTML = "";

      if (!wx) {
        el.innerHTML = `
          <div class="weather-top">
            <div>
              <div class="weather-code">${titleText}</div>
              <div class="weather-station">No weather data</div>
            </div>
          </div>
          <div class="weather-window">No weather information returned.</div>
        `;
        return;
      }

      const tafText = Array.isArray(wx.taf) ? wx.taf.join(" ") : safeValue(wx.taf, "(none)");

      el.innerHTML = `
        <div class="weather-top">
          <div>
            <div class="weather-code">${titleText}</div>
            <div class="weather-station">${safeValue(wx.name)}</div>
          </div>
          <div class="weather-code">${joinParts([wx.iata, wx.icao], " / ")}</div>
        </div>
        <div class="weather-window">${safeValue(wx.window_label, "Weather window not available.")}</div>
        <div class="wx-block">
          <div class="wx-label">METAR</div>
          <div class="wx-value">${safeValue(wx.metar, "(none)")}</div>
        </div>
        <div class="wx-block">
          <div class="wx-label">TAF</div>
          <div class="wx-value">${tafText || "(none)"}</div>
        </div>
      `;
    }

    function renderNotamCard(el, titleText, airport, items) {
      const airportName = airport?.name || "No airport data";
      const airportCode = [airport?.iata, airport?.icao].filter(Boolean).join(" / ");
      const list = Array.isArray(items) ? items : [];

      function cleanNotamText(notam) {
        let text = safeValue(notam?.raw_text, "(no text)");
        const notamId = safeValue(notam?.notam_id, "").trim();

        if (notamId) {
          const escapedId = notamId.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
          text = text.replace(new RegExp("^" + escapedId + "\\s*", "i"), "");
          text = text.replace(new RegExp("\\b" + escapedId + "\\b\\s*(?=COMPANY NOTAM\\s*-)", "i"), "");
          text = text.replace(new RegExp("\\b" + escapedId + "\\b\\s*(?=-\\s*CHART NOTAM)", "i"), "");
        }

        text = text.replace(
          /^VALID:?\s*\d{10}\s*-\s*(?:\d{10}|UFN|PERM)(?:\s+EST)?\s*/i,
          ""
        );

        text = text.replace(
          /^-\s*(?:CHART NOTAM|COMPANY NOTAM)\s+VALID:?\s*\d{10}\s*-\s*(?:\d{10}|UFN|PERM)(?:\s+EST)?\s*/i,
          ""
        );

        text = text.replace(/^(?:CHART NOTAM|COMPANY NOTAM)\s*-\s*/i, "");
        text = text.replace(/^(?:CHART NOTAM|COMPANY NOTAM)\s+/i, "");
        text = text.replace(/^\-\s*/, "");
        text = text.trim();

        return text || "(no text)";
      }

      if (!list.length) {
        el.innerHTML = `
          <div class="notam-top">
            <div>
              <div class="notam-code">${titleText}</div>
              <div class="notam-station">${airportName}</div>
            </div>
            <div class="notam-code">${airportCode || "-"}</div>
          </div>
          <div class="notam-empty">No important NOTAMs found.</div>
        `;
        return;
      }

      el.innerHTML = `
        <div class="notam-top">
          <div>
            <div class="notam-code">${titleText}</div>
            <div class="notam-station">${airportName}</div>
          </div>
          <div class="notam-code">${airportCode || "-"}</div>
        </div>
        <div class="notam-list">
          ${list.map(notam => `
            <div class="notam-item">
              <div class="notam-meta">
                <span class="notam-badge">${safeValue(notam.notam_id, "NOTAM")}</span>
                <span class="notam-badge">${safeValue(notam.validity, "Validity unavailable")}</span>
              </div>
              <div class="notam-text">${cleanNotamText(notam)}</div>
            </div>
          `).join("")}
        </div>
      `;
    }

    uploadBtn.addEventListener("click", async () => {
      errorEl.textContent = "";
      resultsEl.style.display = "none";
      emptyStateEl.style.display = "grid";
      detailsGrid.innerHTML = "";
      dispatchNotesGrid.innerHTML = "";
      depWeatherCard.innerHTML = "";
      destWeatherCard.innerHTML = "";
      altWeatherCard.innerHTML = "";
      depNotamsCard.innerHTML = "";
      destNotamsCard.innerHTML = "";
      altNotamsCard.innerHTML = "";

      const file = fileInput.files[0];

      if (!file) {
        errorEl.textContent = "Please choose a PDF file first.";
        return;
      }

      uploadBtn.disabled = true;
      uploadBtn.textContent = "Generating...";
      statusPill.textContent = "Information only";
      statusCopy.textContent = "Info only. Always verify against the full OFP.";

      try {
        const formData = new FormData();
        formData.append("file", file);

        const res = await fetch("/api/briefing", {
          method: "POST",
          body: formData
        });

        if (!res.ok) {
          const text = await res.text();
          throw new Error(`Server error ${res.status}: ${text}`);
        }

        const data = await res.json();
        const strip = data.strip || {};

        renderSummary(strip);
        renderDetails(strip);
        renderDispatchNotes(data.dispatch_notes);
        renderWeatherCard(depWeatherCard, "Departure", data.dep_weather);
        renderWeatherCard(destWeatherCard, "Destination", data.dest_weather);
        renderWeatherCard(altWeatherCard, "Alternate", data.alt_weather);

        renderNotamCard(depNotamsCard, "Departure", data.dep_weather, data.dep_notams);
        renderNotamCard(destNotamsCard, "Destination", data.dest_weather, data.dest_notams);
        renderNotamCard(altNotamsCard, "Alternate", data.alt_weather, data.alt_notams);

        emptyStateEl.style.display = "none";
        resultsEl.style.display = "grid";
      } catch (err) {
        console.error(err);
        errorEl.textContent = err.message;
      } finally {
        uploadBtn.disabled = false;
        uploadBtn.textContent = "Generate Brief";
      }
    });

    resetUI();
  </script>
